def handle_non_numeric_data(new_df):
    columns = new_df.columns.values
    for column in columns:
        text_digit_vals ={}
        
        def convert_to_int(val):
            return text_digit_vals[val]
        
        if new_df[column].dtype != np.int64 and new_df[column].dtype != np.float64:
            column_contents = new_df[column].values.tolist()
            unique_elements = set(column_contents)
            x=0
            for unique in unique_elements:
                if unique not in text_digit_vals:
                    text_digit_vals[unique] = x
                    x+=1
            new_df[column] = list(map(convert_to_int, new_df[column]))
            
    return new_df

new_df = handle_non_numeric_data(new_df)
